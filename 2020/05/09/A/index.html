一道英语阅读题

甚至花了20多分钟才ac

#include<bits/stdc++.h>
#define MAX 400005
#define INF 0x3f3f3f3f
using namespace std;

int m, n;
int maxx[MAX], minn[MAX], pos[MAX] ,idx[MAX];

int main(){

cin >> n >> m;
for (int i=1; i<=n; i++){
    maxx[i] = i;
    minn[i] = i;
    idx[i] = i;
    pos[i] = i;
}

for(int i=1; i<=m; i++){
    int x;
    scanf("%d", &x);
    if(x == pos[1]) continue;
    int p = idx[x];
    swap(pos[p], pos[idx[x]-1]);
    idx[x] -= 1;
    idx[pos[p]] = p;
    maxx[pos[p]] = max(p, maxx[pos[p]]);
    minn[x] = min(minn[x], p - 1);

}

for(int i=1; i<=n; i++){
    cout << minn[i] << " " << maxx[i] << '\n';
}

}
